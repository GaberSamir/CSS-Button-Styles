Hover Animation Button
/* Button with slide-in animation on hover */
.hover-animation-btn {
  background-color: #2ecc71; /* Green background */
  color: white; /* White text color */
  border: none; /* Remove default border */
  padding: 15px 30px; /* Vertical and horizontal padding */
  font-size: 1.1rem; /* Font size */
  border-radius: 5px; /* Slightly rounded corners */
  cursor: pointer; /* Pointer cursor on hover */
  position: relative; /* Positioning context for pseudo-element */
  overflow: hidden; /* Hide overflow from pseudo-element */
  z-index: 1; /* Ensure text is above pseudo-element */
}

.hover-animation-btn::before {
  content: ''; /* Empty content for pseudo-element */
  position: absolute; /* Absolute positioning */
  top: 0; /* Position from top */
  left: 0; /* Position from left */
  width: 100%; /* Full width */
  height: 100%; /* Full height */
  background-color: #27ae60; /* Darker green */
  transform: translateX(-100%); /* Start from left outside */
  transition: transform 0.3s; /* Smooth transition for transform */
  z-index: -1; /* Behind the text */
}

.hover-animation-btn:hover::before {
  transform: translateX(0); /* Slide in on hover */
}
